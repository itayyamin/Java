Q4: Count Down from a Number to 1
A4:
        read
        store n

    LOOP:
        load n
        gotoz END       // if n == 0, stop

        load n
        write           // write current n

        load n          // you can't count on D to still be n afterward, especially if other commands (like load, write
        sub one
        store n         // n = n - 1

        goto LOOP       // repeat

    END:
        stop


 Q5: sum of first n numbers
 A5:
     read
     store n

     load zero
     store sum

     load one
     store i         // i = 1

 LOOP:
     load n
     sub i
     gotop CONTINUE  // if i <= n

     goto DONE       // if i > n

 CONTINUE:
     load sum
     add i
     store sum       // sum += i

     load i
     add one
     store i         // i++

     goto LOOP

 DONE:
     load sum
     write
     stop

Q2:

    read
    store v         // Store base value v
    load zero
    store sum       // Initialize sum = 0

LOOP:
    read
    store x         // Read next number x
    load x
    sub zero
    gotoz END       // If x == 0, end

    // Compute x - v
    load x
    sub v
    store diff

    // If diff > 0, use it directly
    load diff
    gotop POS

    // Else: negate it to get abs value
    load zero
    sub diff
    store abs
    goto ADD

POS:
    load diff
    store abs

ADD:
    load sum
    add abs
    store sum
    goto LOOP

END:
    load sum
    write
    stop
